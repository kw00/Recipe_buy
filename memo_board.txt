--게시판 테이블
drop table board;

create table board( 
	num number not null primary key,
	writer varchar2(50) not null,
	email varchar2(50),
	title varchar2(50) not null,
	content varchar2(3000) not null,
	inputdate date default sysdate,
	views int default 0
);

drop sequence boardseq;
create sequence boardseq
  increment by 1
  start with 1
  minvalue 1
  maxvalue 10000
  nocache;
